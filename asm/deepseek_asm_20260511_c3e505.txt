;==============================================================================
; ZELIARD ADLIB MUSIC DRIVER (YM3812 / OPL2)
;==============================================================================
; This driver provides multi‑channel music playback on the AdLib OPL2
; synthesizer.  It is called periodically via interrupt 60h and processes up
; to 6 melodic channels + 1 rhythm/percussion channel.  The event language
; is a compact command/note stream similar to the sound‑effect driver.
;
; OPL2 I/O:
;   Status/Index port   388h
;   Data port           389h
;   Register write implementation: see OPL_WriteReg (formerly sub_B08)
; All writes are guarded by a channel mute flag (sub_B00 / Ch_MuteCheck).
;
; Channel state structure (size 44 bytes):
MusicCh_State   struc
seq_ptr         dw ?    ; 0x00 ; pointer to next event in sequence
seq_save        dw ?    ; 0x02 ; saved seq_ptr (used by sub‑routines)
dur_tbl_save    dw ?    ; 0x04 ; saved dur_tbl_base (for sub‑routines)
loop_cnt        db ?    ; 0x06 ; loop counter (unused in melodic channels)
                db ?    ; 0x07
                db ?    ; 0x08
                db ?    ; 0x09
dur_tbl_base    dw ?    ; 0x0A ; pointer to duration multiplier table
opl_channel     db ?    ; 0x0C ; OPL channel number (0‑5 melodic, 6 rhythm)
flags           db ?    ; 0x0D ; bit 0: track finished, bit 6: envelope enabled
note_timer      db ?    ; 0x0E ; countdown for current event
volume          db ?    ; 0x0F ; total level (0‑7Fh, higher = quieter)
octave_shift    db ?    ; 0x10 ; octave / block multiplier (0..7)
flags2          db ?    ; 0x11 ; bit 1: envelope active, bit 4: pause, bit 5: vibrato active, etc.
note_dur_init   db ?    ; 0x12 ; initial note duration (from sequence)
algo_feedback   db ?    ; 0x13 ; low nibble = register C0h value (feedback/algorithm)
default_dur     db ?    ; 0x14 ; default note duration (used by rhythm channel)
perc_mask       db ?    ; 0x15 ; percussion instrument mask (rhythm channel only)
fnum_block      dw ?    ; 0x16 ; frequency: upper 5 bits = block, lower 10 bits = F‑number
transpose       db ?    ; 0x18 ; semitone transpose
pitch_bend_acc  dw ?    ; 0x19 ; pitch bend accumulator
pitch_bend_frac db ?    ; 0x1A ; fractional part of pitch bend
env_countdown   db ?    ; 0x1B ; envelope step counter
env_step_up     dw ?    ; 0x1C ; envelope increment (direction up)
env_step_down   dw ?    ; 0x1E ; envelope decrement (direction down)
env_scale       db ?    ; 0x20 ; envelope multiplier/scale (derived from note freq)
env_hold        db ?    ; 0x21 ; hold counter (from envelope data)
env_par1        db ?    ; 0x22 ; envelope parameter #1
env_par2        db ?    ; 0x23 ; envelope parameter #2
env_par3        db ?    ; 0x24 ; envelope parameter #3
env_par4        db ?    ; 0x25 ; envelope parameter #4
env_flags       db ?    ; 0x26 ; bit 7: alternate; lower 5 bits: period divider
env_alt_count   db ?    ; 0x27 ; toggle counter for par3/par4
instr_ptr       dw ?    ; 0x28 ; pointer to current instrument definition (15 bytes)
                db ?    ; 0x2A ; unused?
channel_mute    db ?    ; 0x2B ; 0 = channel plays, non‑zero = silenced
MusicCh_State   ends

;==============================================================================
include common.inc
include adlib.inc

                .286
                .model small

segment_shift   equ 0ff00h

music_seg       segment byte public 'CODE'
                assume cs:music_seg
                org 100h
start:

; Far entry point called from the game’s timer interrupt.
music_drv_poll_far      proc far
                jmp     near ptr sub_3E5
music_drv_poll_far      endp


; ===========================================================================
; Interrupt 60h handler (new).  Dispatches to driver API functions based
; on AX (function 0..7), with additional parameters in CL.
; ===========================================================================
int60_new       proc far
                push    ax
                push    bx
                push    cx
                push    dx
                push    di
                push    si
                push    bp
                push    ds
                push    es
                push    cs
                pop     es
                cld
                cmp     ax, 8
                jnb     short loc_11D        ; invalid function number
                mov     bx, ax
                shl     bx, 1
                call    cs:music_funcs[bx]
loc_11D:
                pop     es
                pop     ds
                pop     bp
                pop     si
                pop     di
                pop     dx
                pop     cx
                pop     bx
                pop     ax
                iret
int60_new       endp

; ---------------------------------------------------------------------------
music_funcs     dw offset music_fn0  ; 0: Init/Play
                dw offset music_fn1  ; 1: Stop
                dw offset music_fn2  ; 2: BGM Toggle (Pause/Resume music)
                dw offset music_fn3  ; 3: SFX Toggle (Pause/Resume sound FX)
                dw offset music_fn4  ; 4: Raw Write (AH=reg, AL=data)
                dw offset music_fn5  ; 5: Drum Config (percussion setup)
                dw offset music_fn6  ; 6: Fade/Mute (channel mute & fade out)
                dw offset music_fn7  ; 7: Master Vol (global attenuation)

; =============== S U B R O U T I N E =======================================
; fn2: BGM Toggle – Pauses or Resumes the main music track.
;   Input: CL = 00h → Pause, CL != 00h → Resume
; ===========================================================================
music_fn2       proc near
                or      cl, cl
                jnz     short loc_145
                call    sub_3D2             ; silence all OPL channels
                mov     byte ptr cs:(music_channel_param - segment_shift), 0FFh ; set music pause flag
                retn
loc_145:
                mov     byte ptr cs:(music_channel_param - segment_shift), 0    ; clear pause
                jmp     loc_386             ; re‑load instruments and resume all channels
music_fn2       endp

; =============== S U B R O U T I N E =======================================
; fn3: SFX Toggle – Pauses or Resumes sound effects / secondary tracks.
;   Input: CL = 00h → Pause, CL != 00h → Resume
; ===========================================================================
music_fn3       proc near
                or      cl, cl
                jz      short loc_15C
                call    sub_3D2             ; silence all
                mov     byte ptr cs:(byte_FF0B - segment_shift), 0FFh ; set SFX pause flag
                retn
loc_15C:
                mov     byte ptr cs:(byte_FF0B - segment_shift), 0    ; clear pause
                jmp     loc_386
music_fn3       endp

; =============== S U B R O U T I N E =======================================
; fn4: Raw Write – Writes a byte directly to OPL2 Register 0.
;   Input: CL = data byte
; ===========================================================================
music_fn4       proc near
                xor     ah, ah              ; register 0
                jmp     word ptr cs:OPL_save_al_to_reg_ah_proc
music_fn4       endp

; =============== S U B R O U T I N E =======================================
; fn5: Drum Config – Sets up OPL2 Rhythm Mode and percussion bits.
;   Input: CL = percussion flags (bits 2‑4 correspond to bass drum, snare,
;          tom, cymbal, hi‑hat).  If no percussion is selected, rhythm mode
;          is disabled.
;   OPL register 0xBD (bits 0‑4: instruments, bit 5: rhythm enable).
; ===========================================================================
music_fn5       proc near
                test    cl, 38h             ; mask for bass,snare,tom,cym,hat (bits 3,4,5)
                jnz     short loc_178
                mov     al, 0FFh            ; value to disable rhythm? (0xBD bit5 = 0)
                call    word ptr cs:OPL_save_al_to_reg_ah_proc
loc_178:
                mov     ch, cl              ; save flags
                mov     dl, 3               ; process 3 percussion instruments? (bass, snare, tom?)
                xor     dh, dh              ; instrument index 0..2
loc_17E:
                shr     ch, 1
                sbb     al, al              ; al = FFh if bit was set, else 00h
                jnz     short loc_199
                mov     al, dh
                add     al, al
                inc     al                  ; al = index*2 + 1
                add     al, al
                add     al, al
                add     al, al
                add     al, al              ; shift to correct bit position in register 0xBD
                or      al, 8Fh             ; set rhythm enable (bit5) and depth bits (bits 6‑7?)
                call    word ptr cs:OPL_save_al_to_reg_ah_proc
loc_199:
                inc     dh
                dec     dl
                jnz     short loc_17E
                retn
music_fn5       endp

; =============== S U B R O U T I N E =======================================
; fn6: Fade/Mute – Manages channel fading and selective muting.
;   If CL = 0: perform a global fade‑out of channels 4 and 5 (heartbeat)
;   and optionally silence other channels depending on pause flags.
;   If CL <> 0: set the mute flags for channels 4 and 5 according to
;   the bits in CL (bit0 = ch4, bit1 = ch5).
; ===========================================================================
music_fn6       proc near
                or      cl, cl
                jnz     short loc_1F6
                ; CL = 0: global operation
                mov     di, offset opl_buf4
                test    byte ptr es:[di+2Bh], 0FFh   ; check ch4 mute
                jz      short loc_1B1
                call    Unmute_Channel                 ; unmute it
loc_1B1:
                add     di, 44                          ; move to opl_buf5
                test    byte ptr es:[di+2Bh], 0FFh
                jz      short loc_1BE
                call    Unmute_Channel
loc_1BE:
                test    byte ptr es:(music_status_flag - segment_shift), 0FFh
                jz      short loc_1C9
                jmp     sub_3D2                         ; stop flag set → silence everything
loc_1C9:
                test    byte ptr es:(byte_FF0B - segment_shift), 0FFh   ; SFX paused?
                jz      short loc_1D4
                jmp     sub_3D2
loc_1D4:
                test    byte ptr es:(music_channel_param - segment_shift), 0FFh ; BGM paused?
                jz      short loc_1DF
                jmp     sub_3D2
loc_1DF:
                mov     ds, es:word_CBF    ; music data segment
                mov     al, 4
                call    sub_3B1            ; re‑init channel 4
                mov     al, 5
                jmp     sub_3B1            ; re‑init channel 5
; ---------------------------------------------------------------------------
loc_1F6:
                mov     di, offset opl_buf4
                shr     cl, 1
                sbb     al, al             ; al = FFh if bit0 set, else 0
                mov     es:[di+2Bh], al    ; set mute flag for channel 4
                add     di, 2Ch            ; size of opl_buf4 is 44 bytes (0x2C), move to opl_buf5
                shr     cl, 1
                sbb     al, al
                mov     es:[di+2Bh], al    ; set mute flag for channel 5
                retn
music_fn6       endp

; =============== S U B R O U T I N E =======================================
; Unmute_Channel – Clears the mute flag and re‑loads the channel’s volume.
;   Input: ES:DI → channel state
; ===========================================================================
Unmute_Channel  proc near
                mov     byte ptr es:[di+2Bh], 0   ; clear mute
                jmp     loc_778                  ; update frequency/volume
Unmute_Channel  endp

; =============== S U B R O U T I N E =======================================
; fn7: Master Vol – Adjusts global attenuation/volume.
;   Input: CL = desired volume (0‑63), larger = louder
;   The value is inverted and stored in `master_volume`.  It is also
;   written to OPL register 07h (this register is not normally used for
;   volume on YM3812; it may be a dummy write that the game tracks).
; ===========================================================================
music_fn7       proc near
                not     cl
                mov     cs:(master_volume - segment_shift), cl
                mov     al, cl
                and     al, 3Fh
                mov     ah, 7
                jmp     word ptr cs:OPL_save_al_to_reg_ah_proc
music_fn7       endp

; =============== S U B R O U T I N E =======================================
; Clear all driver variables and channel buffers, set stop flag.
; ===========================================================================
sub_21F         proc near
                cld
                push    cs
                pop     es
                mov     di, offset opl_buf0
                mov     cx, 302             ; total size of all channel buffers (6*44 + 1*44? Actually 6*44=264 for melodic, rhythm=44, total 308, but 302 cover them)
                xor     al, al
                rep stosb
                mov     es:(byte_FF21 - segment_shift), al
                mov     es:(byte_FF22 - segment_shift), al
                mov     es:(byte_FF23 - segment_shift), al
                mov     es:(byte_FF24 - segment_shift), al
                mov     es:(byte_FF25 - segment_shift), al
                not     al
                mov     es:(music_status_flag - segment_shift), al  ; set stop flag (non‑zero)
                retn
sub_21F         endp

; =============== S U B R O U T I N E =======================================
; fn0: Init/Play – Clears buffers, loads new music data, and starts playback.
;   Input: ES:SI → music data header (in the data segment pointed by word_CBF)
; ===========================================================================
music_fn0       proc near
                call    sub_21F             ; clear all state
                call    sub_3D2             ; silence OPL
                mov     ax, 120h
                call    OPL_WriteReg        ; write dummy value to register 0? (0x12 → reg0)
                mov     es:music_seq_ptr, si
                mov     es:word_CBF, ds    ; save music data segment
                mov     dx, si              ; base address for relative pointers
                inc     si                  ; skip first byte (music format header)
                mov     di, offset opl_buf0
                mov     cx, 6               ; six melodic channels
                xor     bl, bl              ; channel index 0..5
loc_268:
                call    Ch_Init_Melodic
                add     di, 44              ; next channel state buffer
                inc     bl
                loop    loc_268
                add     si, 6               ; skip header data
                ; Set up rhythm channel (opl_buf6)
                mov     di, offset opl_buf6
                call    Ch_Init_Base        ; load seq pointer
                mov     byte ptr es:[di+0Ch], 80h ; flag: rhythm channel (bit7 = 1)
                lodsw                       ; pointer to instrument table
                add     ax, dx
                mov     es:word_CC1, ax
                lodsw                       ; (unknown)
                lodsw                       ; pointer to duration table for rhythm
                add     ax, dx
                mov     es:word_CC3, ax
                mov     ax, 0BD20h
                call    OPL_WriteReg        ; enable rhythm mode (bit5) with no percussion instruments yet? (depth=0)
                xor     al, al
                mov     es:24h, al          ; clear some scratch variables? (overlapping with code, but this segment is data area)
                mov     es:26h, al
                mov     es:byte_CBB, al
                mov     es:byte_CBC, al
                inc     al
                mov     es:byte_CB9, al
                mov     es:byte_CC6, al
                mov     es:byte_CB7, 7Fh    ; TempoAccum = default ~ mid tempo
                mov     es:byte_CB8, 0
                mov     es:byte_CC5, 0
                push    cs
                pop     ds
                assume ds:music_seg
                ; Load default instruments for channels 6,7,8 (rhythm melodic operators)
                mov     si, offset byte_B7E
                mov     bl, 6
                call    Load_Rhythm_Instrument
                mov     bl, 7
                call    Load_Rhythm_Instrument
                mov     bl, 8
                call    Load_Rhythm_Instrument
                ; Set default frequencies for percussion instruments
                mov     cx, 520h
                add     ah, 0A6h
                mov     al, cl
                call    OPL_WriteReg
                mov     ah, 0B6h
                mov     al, ch
                call    OPL_WriteReg
                mov     cx, 550h
                mov     ah, 0A7h
                mov     al, cl
                call    OPL_WriteReg
                add     ah, 0B7h
                mov     al, ch
                call    OPL_WriteReg
                mov     cx, 3C0h
                mov     ah, 0A8h
                mov     al, cl
                call    OPL_WriteReg
                add     ah, 0B8h
                mov     al, ch
                jmp     OPL_WriteReg
music_fn0       endp

; =============== S U B R O U T I N E =======================================
; Load a standard instrument for the given OPL channel (BL = 6,7,8).
; The instrument data is at SI (9 bytes + 4?).
; ===========================================================================
Load_Rhythm_Instrument proc near
                push    bx
                xor     bh, bh
                mov     ah, byte_B75[bx]    ; operator base number
                add     ah, 20h             ; register 0x20+op (AM/VIB/EG/KSR/MULT)
                mov     cx, 4
loc_31A:
                lodsb
                call    OPL_WriteReg
                add     ah, 3               ; next operator
                lodsb
                call    OPL_WriteReg
                add     ah, 1Dh             ; skip to next register block (0x60+op)
                loop    loc_31A
                add     ah, 40h             ; now 0xE0+op? (Wave Select)
                lodsb
                rol     al, 1
                rol     al, 1
                call    OPL_WriteReg
                add     ah, 3
                rol     al, 1
                rol     al, 1
                call    OPL_WriteReg
                pop     bx
                mov     ah, bl
                ror     al, 1
                ror     al, 1
                ror     al, 1
                ror     al, 1
                add     ah, 0C0h            ; feedback/algorithm for channel BL
                jmp     OPL_WriteReg
Load_Rhythm_Instrument endp

; =============== S U B R O U T I N E =======================================
; Initialise a single melodic channel state.
;   DI → channel state, BL = channel number (0‑5)
; ===========================================================================
Ch_Init_Melodic proc near
                call    Ch_Init_Base        ; load seq pointer, set note_timer=1
                mov     byte ptr es:[di+10h], 3     ; default octave shift
                mov     es:[di+0Ch], bl             ; OPL channel number
                mov     byte ptr es:[di+12h], 1     ; initial note duration
                mov     byte ptr es:[di+0Fh], 7Fh   ; volume (max attenuation = silent)
                mov     byte ptr es:[di+18h], 0     ; transpose = 0
                and     byte ptr es:[di+0Dh], 0BFh  ; clear envelope enable
                mov     byte ptr es:[di+11h], 0     ; clear flags2
                retn
Ch_Init_Melodic endp

; =============== S U B R O U T I N E =======================================
; Common base initialisation for any channel:
; load the sequence pointer from the music data (2 bytes), store as both
; current and saved pointer, and set note_timer to 1.
;   Input: DS:SI → music data, DX = base offset
;   DI → channel state
; ===========================================================================
Ch_Init_Base    proc near
                lodsw                       ; get relative pointer
                add     ax, dx              ; make absolute
                mov     es:[di], ax         ; seq_ptr
                mov     es:[di+2], ax       ; seq_save
                mov     byte ptr es:[di+0Eh], 1 ; note_timer = 1
                retn
Ch_Init_Base    endp

; ---------------------------------------------------------------------------
; Re‑load base instruments and re‑start all active channels (used after
; toggling BGM/SFX back on).
loc_386:
                push    cs
                pop     ds
                mov     si, offset byte_B7E
                mov     bl, 6
                call    Load_Rhythm_Instrument
                mov     bl, 7
                call    Load_Rhythm_Instrument
                mov     bl, 8
                call    Load_Rhythm_Instrument
                ; Update volumes on rhythm channel
                mov     di, offset opl_buf6
                call    Rhythm_UpdateVolumes
                ; Re‑init all 6 melodic channels
                xor     al, al
                mov     cx, 6
loc_3A5:
                push    cx
                push    ax
                call    Ch_Reinit_From_MusicData
                pop     ax
                inc     al
                pop     cx
                loop    loc_3A5
                retn

; =============== S U B R O U T I N E =======================================
; Re‑initialise a melodic channel from the current music data.
;   Input: AL = channel index (0..5)
; ===========================================================================
Ch_Reinit_From_MusicData proc near
                push    cs
                pop     es
                mov     ds, es:word_CBF    ; music data segment
                mov     cl, 44
                mul     cl
                add     ax, offset opl_buf0
                mov     di, ax             ; DI → channel state
                mov     si, es:[di+29h]    ; get current instrument pointer
                jmp     sub_5AF            ; reload instrument and update volumes
Ch_Reinit_From_MusicData endp

; =============== S U B R O U T I N E =======================================
; fn1: Stop – Silences all channels and halts the driver.
; ===========================================================================
music_fn1       proc near
                call    sub_3D2
                mov     byte ptr es:(music_status_flag - segment_shift), 0FFh
                retn
music_fn1       endp

; =============== S U B R O U T I N E =======================================
; Silence all OPL channels (key‑off, disable rhythm, volume zero).
; ===========================================================================
sub_3D2         proc near
                mov     ax, 0BD00h
                call    OPL_WriteReg        ; rhythm disable, all instruments off
                mov     cx, 9
                mov     ah, 0B0h            ; register 0xB0..0xB8: key‑on/block
loc_3DD:
                call    OPL_WriteReg        ; write 0 → key off
                inc     ah
                loop    loc_3DD
                retn
sub_3D2         endp

; ===========================================================================
; Main per‑tick processing.  Called from the far poll procedure.
; Advances the six melodic channels and the rhythm channel.
; ===========================================================================
sub_3E5         proc far
                cld
                call    sub_3EA
                retf
sub_3E5         endp

sub_3EA         proc near
                test    byte ptr cs:(music_status_flag - segment_shift), 0FFh
                jz      short loc_3F3
                retn                            ; driver stopped
loc_3F3:
                test    byte ptr cs:(byte_FF0B - segment_shift), 0FFh
                jz      short loc_3FC
                retn                            ; SFX paused, don't update music
loc_3FC:
                dec     cs:byte_CC6             ; tick divider
                jz      short loc_404
                retn                            ; not yet time for this tick
loc_404:
                mov     cs:byte_CC6, 2          ; re‑arm counter (every 2 interrupts)
                push    cs
                pop     es
                mov     ds, es:word_CBF
                test    byte ptr es:(byte_FF24 - segment_shift), 0FFh ; fade requested?
                jz      short loc_425
                call    sub_463                 ; process fade step
                test    byte ptr es:(music_status_flag - segment_shift), 0FFh
                jz      short loc_425
                retn
loc_425:
                not     es:byte_CBB             ; toggle envelope/vibrato phase?
                mov     al, es:byte_CB7          ; TempoAccum
                add     es:byte_CB8, al          ; add to fractional tempo counter
                sbb     al, al
                mov     es:byte_CBA, al          ; store carry (tempo tick flag)
                ; Process all seven channels (0‑5 melodic, 6 rhythm)
                mov     di, offset opl_buf0
                call    Ch_Sequencer_Melodic
                mov     di, offset opl_buf1
                call    Ch_Sequencer_Melodic
                mov     di, offset opl_buf2
                call    Ch_Sequencer_Melodic
                mov     di, offset opl_buf3
                call    Ch_Sequencer_Melodic
                mov     di, offset opl_buf4
                call    Ch_Sequencer_Melodic
                mov     di, offset opl_buf5
                call    Ch_Sequencer_Melodic
                mov     di, offset opl_buf6
                jmp     loc_8A9                 ; rhythm channel sequencer
sub_3EA         endp

; ===========================================================================
; Fade processor – gradually increases the attenuation in byte_FF25 until
; it overflows, then stops the driver completely.
; ===========================================================================
sub_463         proc near
                mov     es:byte_CBC, 0
                dec     es:byte_CB9             ; fade speed counter
                jz      short loc_471
                retn
loc_471:
                mov     al, es:(byte_FF24 - segment_shift)  ; fade speed reload
                mov     es:byte_CB9, al
                mov     es:byte_CBC, 0FFh                   ; flag: fade step done
                mov     al, es:(byte_FF25 - segment_shift)  ; current attenuation
                add     al, 4                               ; increase (quieter)
                jnb     short loc_498
                ; Attenuation overflow → stop
                mov     byte ptr es:(byte_FF24 - segment_shift), 0
                mov     byte ptr es:(music_status_flag - segment_shift), 0FFh
                call    sub_3D2
                mov     al, 0FFh
loc_498:
                mov     es:(byte_FF25 - segment_shift), al
                retn
sub_463         endp

; ===========================================================================
; Channel sequencer for melodic channels (0‑5).  Interprets the event
; stream and handles notes, commands, envelopes, and volume updates.
; ===========================================================================
Ch_Sequencer_Melodic proc near
                test    byte ptr es:[di+0Dh], 1     ; track finished?
                jz      short loc_4A5
                retn
loc_4A5:
                mov     ax, offset sub_7A6          ; return address for envelope processing
                push    ax
                test    es:byte_CBA, 0FFh           ; tempo tick?
                jz      short loc_4B2
                retn                                 ; no tick this frame
loc_4B2:
                dec     byte ptr es:[di+0Eh]        ; note_timer--
                jz      short loc_4CE               ; timer expired → fetch event
                mov     al, es:[di+12h]             ; note_dur_init
                cmp     al, es:[di+0Eh]
                jnb     short loc_4C3               ; still in sustain phase
                retn
loc_4C3:
                test    byte ptr es:[di+11h], 10h   ; pause flag?
                jz      short loc_4CB
                retn
loc_4CB:
                jmp     loc_778                     ; still playing → update freq/vol
; --- fetch next event ---
loc_4CE:
                mov     si, es:[di]                 ; seq_ptr
loc_4D1:
                lodsb
                or      al, al
                js      short loc_4D9               ; high bit set → command
                jmp     loc_6E8                     ; note event
loc_4D9:
                mov     bx, offset loc_4D1
                push    bx                          ; most commands return to loop
                test    al, 40h
                jnz     short loc_4E4
                jmp     loc_598                     ; group2: instrument change (0x80‑0xBF)
            ; --- Command group 1 (0xC0‑0xFF) ---
loc_4E4:
                cmp     al, 0D0h
                jnb     short loc_4EB
                jmp     loc_699                     ; 0xC0..0xCF: volume relative
loc_4EB:
                cmp     al, 0D8h
                jnb     short loc_4F2
                jmp     loc_6CC                     ; 0xD0..0xD7: set octave
loc_4F2:
                cmp     al, 0xE0h
                jnb     short loc_4F9
                jmp     loc_6D3                     ; 0xD8..0xDF: set duration
loc_4F9:
                and     al, 1Fh
                add     al, al
                mov     bl, al
                xor     bh, bh
                jmp     es:off_506[bx]              ; 0xE0..0xFF dispatch table
Ch_Sequencer_Melodic endp
